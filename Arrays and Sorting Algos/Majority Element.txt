int main()
{

    int a[] = {2, 4, 7, 8, 9, 10};

    int n = sizeof(a) / sizeof(a[0]);

    sort(a, n); // sorting the array.
    int i;
    for (i = 0; i <= n - 2; i += 2)
    {
        swapInt(&a[i], &a[i + 1]); // swapping alternate elements
    }
    for (i = 0; i < n; i++)
    {
        cout << a[i] << " ";
    }
}
