REVERSE ARRAY O(N) 


int main()
{
    int a[] = {1, 2, 3, 4, 5};
    int n = sizeof(a) / sizeof(a[0]);

    int i = 0, j = n - 1;

    while (i != j)
    {
        swapInt(&a[i], &a[j]);
        i++;
        j--;
    }

    for (i = 0; i < n; i++)
    {
        cout << a[i] << " ";
    }
}
