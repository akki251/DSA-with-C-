// SECOND LARGEST ELEMENT OF ARRAY 

int main()
{
    int a[] = {12, 35, 10, 34, 1};

    int n = sizeof(a) / sizeof(a[0]);

    int i;
    int largest = 0;
    for (i = 0; i < n; i++)
    {
        if (a[i] > a[largest])
            largest = i;
    }

    a[largest] = -1;
    int secondLargest = 0;

    for (i = 0; i < n; i++)
    {
        if (a[i] > a[secondLargest])
            secondLargest = i;
    }

    cout << a[secondLargest];
}
