#include <iostream>
#include <math.h>
using namespace std;


int main()
{

    int a[] = {21, 23, 56, 65, 34, 54, 76, 32, 23, 45, 21, 23, 25};
    int n = sizeof(a) / sizeof(a[0]);

    int k, x;
    cin >> x;
    cin >> k;

    int i, j;

    for (i = 0; i < n; i = i + k)
    {
        for (j = 0; j < k; j ++)
        {
            if (a[i + j] == x)
                break;
        }

        if (j == k)
            cout << 0;
    }

    if (i == n)
        cout << 1;

    for (j = i - k; j < n; j++)
    {
        if (a[j] == x)
            break;
    }

    if (j == n)
        cout << 0;

  cout << 1;
}
