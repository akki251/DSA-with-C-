#include <iostream>
#include <math.h>
using namespace std;

int main()
{

    int a[] = {3, 1, 3, 3, 2};
    int n = sizeof(a) / sizeof(a[0]);
    int i, j;
    int b[n];
    int count = 0;
    for (i = 0; i < n; i++)
    {
        if (a[i] != -1)
        {
            count = 1;
            for (j = i + 1; j < n; j++)
            {
                if (a[i] == a[j])
                {
                    a[j] = -1;
                    count++;
                }
            }
        }

        b[i] = count;
    }

    for (i = 0; i < n; i++)
    {

        if (a[i] != -1)
        {
            if (b[i] > n / 2)
            {
                cout << a[i];
                return 0;
            }
        }
    }

    cout << -1;
}
